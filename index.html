// Anket sonuç sheet'leri — form bağlantısı sonrası otomatik oluşur var sheets = { tuketici: ss.getSheetByName('Tüketici Yanıtları'), bayi: ss.getSheetByName('Bayi Yanıtları'), b2b: ss.getSheetByName('B2B Yanıtları') }; var dashboard = ss.getSheetByName('Dashboard') || ss.insertSheet('Dashboard'); dashboard.clearContents(); // Başlık dashboard.getRange('A1').setValue('MAYAS YO-SU — Anket Dashboard'); dashboard.getRange('A2').setValue('Son güncelleme: ' + new Date().toLocaleString('tr-TR')); var row = 4; Object.keys(sheets).forEach(function(key) { var sheet = sheets[key]; if (!sheet) return; var data = sheet.getDataRange().getValues(); var count = data.length - 1; // başlık satırı çıkar dashboard.getRange(row, 1).setValue(key.toUpperCase() + ' ANKETİ'); dashboard.getRange(row, 2).setValue('Toplam yanıt: ' + count); row += 2; // Basit frekans analizi — ilk 5 soruyu özetle if (data.length > 1) { var headers = data[0]; for (var col = 1; col < Math.min(6, headers.length); col++) { var freq = {}; for (var r = 1; r < data.length; r++) { var val = String(data[r][col]); freq[val] = (freq[val] || 0) + 1; } dashboard.getRange(row, 1).setValue(headers[col]); row++; Object.keys(freq).forEach(function(answer) { dashboard.getRange(row, 2).setValue(answer); dashboard.getRange(row, 3).setValue(freq[answer]); dashboard.getRange(row, 4).setValue( Math.round(freq[answer] / count * 100) + '%' ); row++; }); row++; } } row += 2; }); // Formatla dashboard.getRange('A1').setFontSize(16).setFontWeight('bold'); dashboard.setColumnWidth(1, 300); dashboard.setColumnWidth(2, 250); dashboard.setColumnWidth(3, 80); dashboard.setColumnWidth(4, 80); Logger.log('✅ Dashboard güncellendi: ' + row + ' satır yazıldı.'); } // Her gece otomatik çalıştır (trigger olarak ekle) function createDailyTrigger() { ScriptApp.newTrigger('updateMAYASDashboard') .timeBased() .everyDays(1) .atHour(8) .create(); Logger.log('✅ Günlük trigger oluşturuldu (her sabah 08:00).'); } ``` ### Bu scripti çalıştırmak için: 1. Google Sheets'i aç (anket yanıtlarının aktığı sheet) 2. Uzantılar → Apps Script 3. Kodu yapıştır 4. Önce `createDailyTrigger` çalıştır → trigger kurulur 5. Sonra `updateMAYASDashboard` çalıştır → ilk özet oluşur --- ## BÖLÜM E: Hızlı Referans — Marka Kılavuzu | Unsur | Değer | |-------|-------| | Ana renk | #E8357A (YO-SU pembe) | | İkinci renk | #7B4FC8 (YO-SU mor) | | Açık zemin | #FAF7F2 (krem) | | Koyu zemin | #1A0A12 (derin koyu) | | Yazı tipi | DM Sans (gövde) + DM Serif Display (başlık) | | Slogan | "Mayasında sağlık var." | | İkincil slogan | "Tek hat, iki ürün, sıfır kayıp." | | Web | www.mayasgida.com | | LinkedIn | linkedin.com/in/musdafi |
